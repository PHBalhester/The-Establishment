**Create `app/hooks/useChartSSE.ts`** -- EventSource hook for SSE candle updates.

This hook manages the SSE connection lifecycle with auto-reconnect:

```typescript
"use client";

import { useEffect, useRef, useState, useCallback } from "react";

export type ConnectionStatus = "connected" | "reconnecting" | "disconnected";

export interface CandleSSEUpdate {
  pool: string;
  resolution: string;
  price: number;
  volume: number;
  timestamp: number; // unix seconds
}

export function useChartSSE(
  onUpdate: (update: CandleSSEUpdate) => void,
) {
  const [status, setStatus] = useState<ConnectionStatus>("disconnected");
  const eventSourceRef = useRef<EventSource | null>(null);
  const onUpdateRef = useRef(onUpdate);
  onUpdateRef.current = onUpdate;

  useEffect(() => {
    let reconnectTimeout: ReturnType<typeof setTimeout>;
    let reconnectAttempts = 0;

    function connect() {
      const es = new EventSource("/api/sse/candles");
      eventSourceRef.current = es;

      es.addEventListener("connected", () => {
        setStatus("connected");
        reconnectAttempts = 0;
      });

      es.addEventListener("candle-update", (event) => {
        try {
          const data = JSON.parse(event.data) as CandleSSEUpdate;
          onUpdateRef.current(data);
        } catch { /* ignore malformed data */ }
      });

      es.onerror = () => {
        es.close();
        setStatus("reconnecting");
        // Exponential backoff: 1s, 2s, 4s, 8s, max 30s
        const delay = Math.min(1000 * 2 ** reconnectAttempts, 30_000);
        reconnectAttempts++;
        reconnectTimeout = setTimeout(connect, delay);
      };
    }

    connect();

    return () => {
      clearTimeout(reconnectTimeout);
      eventSourceRef.current?.close();
      setStatus("disconnected");
    };
  }, []);

  return { status };
}
```

**Key design decisions:**
- `onUpdateRef` pattern prevents re-establishing the EventSource when the callback changes (common React pattern for stable effect dependencies)
- Exponential backoff on error prevents hammering the server
- Status exposed for UI indicator (green/amber/red dot)

**Create `app/hooks/useChartData.ts`** -- Manages chart candle state from REST + SSE.

This hook:
1. Fetches historical candles from REST API when pool, resolution, or time range changes
2. Receives real-time updates from useChartSSE and updates the candle array
3. Exposes the candle data array for the chart component

```typescript
"use client";

import { useState, useCallback, useEffect, useRef } from "react";
import { useChartSSE, ConnectionStatus, CandleSSEUpdate } from "./useChartSSE";

export interface CandleData {
  time: number;  // Unix seconds (UTCTimestamp for TradingView)
  open: number;
  high: number;
  low: number;
  close: number;
  volume: number;
}

export type TimeRange = "1H" | "4H" | "1D" | "1W";
export type Resolution = "1m" | "5m" | "15m" | "1h" | "4h" | "1d";

// Map time ranges to sensible default resolutions
const RANGE_TO_RESOLUTION: Record<TimeRange, Resolution> = {
  "1H": "1m",
  "4H": "5m",
  "1D": "15m",
  "1W": "1h",
};

// Map time ranges to seconds of history to fetch
const RANGE_TO_SECONDS: Record<TimeRange, number> = {
  "1H": 3600,
  "4H": 14400,
  "1D": 86400,
  "1W": 604800,
};
```

**The hook signature:**

```typescript
export function useChartData(pool: string, options?: {
  initialRange?: TimeRange;
  initialResolution?: Resolution;
}) {
  const [candles, setCandles] = useState<CandleData[]>([]);
  const [range, setRange] = useState<TimeRange>(options?.initialRange ?? "1D");
  const [resolution, setResolution] = useState<Resolution>(
    options?.initialResolution ?? RANGE_TO_RESOLUTION["1D"]
  );
  const [loading, setLoading] = useState(true);

  // When range changes, auto-select a sensible resolution
  const handleRangeChange = useCallback((newRange: TimeRange) => {
    setRange(newRange);
    setResolution(RANGE_TO_RESOLUTION[newRange]);
  }, []);

  // Fetch historical candles from REST API
  useEffect(() => {
    if (!pool) return;
    setLoading(true);
    const now = Math.floor(Date.now() / 1000);
    const from = now - RANGE_TO_SECONDS[range];
    fetch(`/api/candles?pool=${pool}&resolution=${resolution}&from=${from}&to=${now}&limit=2000`)
      .then(res => res.json())
      .then((data: CandleData[]) => {
        setCandles(data);
        setLoading(false);
      })
      .catch(() => setLoading(false));
  }, [pool, resolution, range]);

  // Handle SSE updates: only process updates for current pool + resolution
  const handleSSEUpdate = useCallback((update: CandleSSEUpdate) => {
    if (update.pool !== pool || update.resolution !== resolution) return;
    // ... (update or append candle)
  }, [pool, resolution]);

  const { status: connectionStatus } = useChartSSE(handleSSEUpdate);

  return {
    candles,
    loading,
    range,
    setRange: handleRangeChange,
    resolution,
    setResolution,
    connectionStatus,
  };
}
```

**SSE update logic for candles:** When a candle-update arrives for the current pool+resolution:
- Find the last candle in the array
- If the update timestamp falls within the last candle's period, update it in-place (new high = max(old high, update price), new low = min(old low, update price), close = update price, volume += update volume)
- If the update timestamp is beyond the last candle, append a new candle
- Use functional setState to avoid stale closure issues

**Pool address mapping:** The hook receives a pool PDA address string (from the pool config constants). The SwapForm's selected pair maps to a pool address via `resolvePool()` from shared/constants.ts.
