**Create `app/db/candle-aggregator.ts`** -- Upserts OHLCV candles for all 6 resolutions.

**Core concept:** When a swap occurs at price P with volume V at time T, we need to update (or create) candles for ALL 6 resolutions simultaneously. Each resolution has a different time bucket:

| Resolution | Bucket Size |
|------------|------------|
| 1m | 60 seconds |
| 5m | 300 seconds |
| 15m | 900 seconds |
| 1h | 3600 seconds |
| 4h | 14400 seconds |
| 1d | 86400 seconds |

**Time truncation:** For a given timestamp and resolution, compute the candle's `openTime` by flooring to the bucket boundary:

```typescript
const RESOLUTION_SECONDS: Record<string, number> = {
  "1m": 60,
  "5m": 300,
  "15m": 900,
  "1h": 3600,
  "4h": 14400,
  "1d": 86400,
};

function truncateToResolution(timestamp: Date, resolution: string): Date {
  const seconds = RESOLUTION_SECONDS[resolution];
  if (!seconds) throw new Error(`Unknown resolution: ${resolution}`);
  const unixSeconds = Math.floor(timestamp.getTime() / 1000);
  const truncated = Math.floor(unixSeconds / seconds) * seconds;
  return new Date(truncated * 1000);
}
```

**Upsert logic:** For each resolution, insert a new candle or update the existing one:

```sql
INSERT INTO candles (pool, resolution, open_time, open, high, low, close, volume, trade_count)
VALUES ($pool, $resolution, $openTime, $price, $price, $price, $price, $volume, 1)
ON CONFLICT (pool, resolution, open_time) DO UPDATE SET
  high = GREATEST(candles.high, EXCLUDED.high),
  low = LEAST(candles.low, EXCLUDED.low),
  close = EXCLUDED.close,  -- latest price becomes close
  volume = candles.volume + EXCLUDED.volume,
  trade_count = candles.trade_count + 1;
```

In Drizzle ORM, this translates to:

```typescript
import { db } from "./connection";
import { candles } from "./schema";
import { sql } from "drizzle-orm";

export interface CandleUpdate {
  pool: string;      // pool PDA address
  price: number;     // derived price (SOL per token)
  volume: number;    // in lamports
  timestamp: Date;   // swap timestamp
}

const RESOLUTIONS = ["1m", "5m", "15m", "1h", "4h", "1d"] as const;

export async function upsertCandles(update: CandleUpdate): Promise<void> {
  // Upsert all 6 resolutions in parallel
  await Promise.all(
    RESOLUTIONS.map((resolution) => {
      const openTime = truncateToResolution(update.timestamp, resolution);
      return db
        .insert(candles)
        .values({
          pool: update.pool,
          resolution,
          openTime,
          open: update.price,
          high: update.price,
          low: update.price,
          close: update.price,
          volume: update.volume,
          tradeCount: 1,
        })
        .onConflictDoUpdate({
          target: [candles.pool, candles.resolution, candles.openTime],
          set: {
            high: sql`GREATEST(${candles.high}, ${update.price})`,
            low: sql`LEAST(${candles.low}, ${update.price})`,
            close: sql`${update.price}`,
            volume: sql`${candles.volume} + ${update.volume}`,
            tradeCount: sql`${candles.tradeCount} + 1`,
          },
        });
    })
  );
}
```

**Return type:** Also export a function `upsertCandlesForSwap` that takes the parsed swap event data and computes price/volume before calling `upsertCandles`, to simplify the webhook handler call site.

**Volume:** Use the SOL-side amount as volume for SOL pools (whether input on buy or output on sell). For PROFIT pools, use the input amount as volume.

**Create `app/lib/sse-manager.ts`** -- In-memory pub/sub for SSE broadcasting.

**Why in-memory:** Railway runs a single Next.js process (no horizontal scaling for devnet). An in-memory Set of subscriber callbacks is sufficient. If we needed multi-process, we'd use Redis pub/sub -- but that's overkill for devnet.

```typescript
export type SSECallback = (data: string) => void;

class SSEManager {
  private subscribers = new Set<SSECallback>();

  subscribe(callback: SSECallback): () => void {
    this.subscribers.add(callback);
    return () => {
      this.subscribers.delete(callback);
    };
  }

  broadcast(event: string, data: unknown): void {
    const payload = `event: ${event}\ndata: ${JSON.stringify(data)}\n\n`;
    for (const callback of this.subscribers) {
      try {
        callback(payload);
      } catch {
        // Subscriber disconnected, will be cleaned up
        this.subscribers.delete(callback);
      }
    }
  }

  get subscriberCount(): number {
    return this.subscribers.size;
  }
}

// Singleton: survives hot reloads via globalThis
const globalForSSE = globalThis as unknown as { sseManager: SSEManager | undefined };
export const sseManager = globalForSSE.sseManager ?? new SSEManager();
if (process.env.NODE_ENV !== "production") {
  globalForSSE.sseManager = sseManager;
}
```

**Create `app/app/api/candles/route.ts`** -- REST API for historical candle data.

**Query parameters:**
- `pool` (required): Pool PDA address (base58 string)
- `resolution` (required): One of "1m", "5m", "15m", "1h", "4h", "1d"
- `from` (optional): ISO timestamp or unix seconds for range start
- `to` (optional): ISO timestamp or unix seconds for range end. Defaults to now.
- `limit` (optional): Max candles to return. Default 500, max 2000.

**Implementation:**

```typescript
import { NextRequest, NextResponse } from "next/server";
import { db } from "@/db/connection";
import { candles } from "@/db/schema";
import { and, eq, gte, lte, asc } from "drizzle-orm";

export const runtime = "nodejs";
export const dynamic = "force-dynamic";

const VALID_RESOLUTIONS = new Set(["1m", "5m", "15m", "1h", "4h", "1d"]);

export async function GET(req: NextRequest) {
  const { searchParams } = req.nextUrl;
  const pool = searchParams.get("pool");
  const resolution = searchParams.get("resolution");
  const from = searchParams.get("from");
  const to = searchParams.get("to");
  const limitStr = searchParams.get("limit");
  // Validate
  if (!pool || !resolution || !VALID_RESOLUTIONS.has(resolution)) {
    return NextResponse.json(
      { error: "pool and resolution (1m|5m|15m|1h|4h|1d) are required" },
      { status: 400 }
    );
  }
  const limit = Math.min(Math.max(parseInt(limitStr || "500", 10) || 500, 1), 2000);
  // Build conditions
  const conditions = [
    eq(candles.pool, pool),
    eq(candles.resolution, resolution),
  ];
  if (from) {
    const fromDate = new Date(isNaN(Number(from)) ? from : Number(from) * 1000);
    conditions.push(gte(candles.openTime, fromDate));
  }
  if (to) {
    const toDate = new Date(isNaN(Number(to)) ? to : Number(to) * 1000);
    conditions.push(lte(candles.openTime, toDate));
  }
  // Query
  const rows = await db
    .select()
    .from(candles)
    .where(and(...conditions))
    .orderBy(asc(candles.openTime))
    .limit(limit);
  // Gap-fill: insert flat candles for missing periods
  const filled = gapFillCandles(rows, resolution, from, to);
  return NextResponse.json(filled);
}
```

**Gap-fill logic:** The CONTEXT.md specifies "carry forward last price (O=H=L=C=last traded price, volume=0)". Implement this in a `gapFillCandles` function:

```typescript
function gapFillCandles(
  rows: typeof candles.$inferSelect[],
  resolution: string,
  from: string | null,
  to: string | null
): Array<{...}> {
  if (rows.length === 0) return [];

  const RESOLUTION_SECONDS: Record<string, number> = {
    "1m": 60, "5m": 300, "15m": 900, "1h": 3600, "4h": 14400, "1d": 86400,
  };
  const step = RESOLUTION_SECONDS[resolution]! * 1000; // ms
  const filled = [];
  let lastPrice = rows[0].close;

  // Determine range
  const rangeStart = from
    ? new Date(isNaN(Number(from)) ? from : Number(from) * 1000).getTime()
    : rows[0].openTime.getTime();
  const rangeEnd = to
    ? new Date(isNaN(Number(to)) ? to : Number(to) * 1000).getTime()
    : rows[rows.length - 1].openTime.getTime();

  let rowIdx = 0;
  for (let t = rangeStart; t <= rangeEnd; t += step) {
    if (rowIdx < rows.length && rows[rowIdx].openTime.getTime() === t) {
      filled.push(formatCandle(rows[rowIdx]));
      lastPrice = rows[rowIdx].close;
      rowIdx++;
    } else {
      // Gap: carry forward last price
      filled.push({
        time: Math.floor(t / 1000), // Unix seconds for TradingView
        open: lastPrice,
        high: lastPrice,
        low: lastPrice,
        close: lastPrice,
        volume: 0,
        tradeCount: 0,
      });
    }
  }
  return filled;
}
```

**Create `app/app/api/sse/candles/route.ts`** -- SSE streaming endpoint.

**Implementation pattern for SSE in Next.js App Router:**

```typescript
import { NextRequest } from "next/server";
import { sseManager } from "@/lib/sse-manager";

export const runtime = "nodejs";
export const dynamic = "force-dynamic";

export async function GET(req: NextRequest) {
  const encoder = new TextEncoder();

  const stream = new ReadableStream({
    start(controller) {
      // Send initial connection event
      controller.enqueue(encoder.encode("event: connected\ndata: {}\n\n"));

      // Subscribe to candle updates
      const unsubscribe = sseManager.subscribe((payload: string) => {
        try {
          controller.enqueue(encoder.encode(payload));
        } catch {
          // Client disconnected
          unsubscribe();
        }
      });

      // Heartbeat: send comment every 15 seconds to keep connection alive
      const heartbeat = setInterval(() => {
        try {
          controller.enqueue(encoder.encode(": heartbeat\n\n"));
        } catch {
          clearInterval(heartbeat);
          unsubscribe();
        }
      }, 15_000);

      // Cleanup on client disconnect
      req.signal.addEventListener("abort", () => {
        clearInterval(heartbeat);
        unsubscribe();
        try { controller.close(); } catch { /* already closed */ }
      });
    },
  });

  return new Response(stream, {
    headers: {
      "Content-Type": "text/event-stream",
      "Cache-Control": "no-cache, no-transform",
      Connection: "keep-alive",
      "X-Accel-Buffering": "no", // Disable nginx/proxy buffering
    },
  });
}
```
