2. Create programs/mock-tax-program/src/lib.rs:
   ```rust
   use anchor_lang::prelude::*;
   use anchor_lang::solana_program::program::invoke_signed;
   use anchor_lang::solana_program::instruction::{AccountMeta, Instruction};

   declare_id!("MockTax1111111111111111111111111111111111");

   pub const SWAP_AUTHORITY_SEED: &[u8] = b"swap_authority";

   #[program]
   pub mod mock_tax_program {
       use super::*;

       /// Forward a swap to the AMM with swap_authority signature.
       /// This is a minimal mock -- real Tax Program would compute taxes.
       ///
       /// The execute_swap instruction:
       /// 1. Derives the swap_authority PDA from this program's ID
       /// 2. Builds the AMM swap instruction with all required accounts
       /// 3. Signs the swap_authority via invoke_signed
       ///
       /// Account ordering for AMM swap instruction (must match exactly):
       /// 0: pool (mut)
       /// 1: vault_a (mut)
       /// 2: vault_b (mut)
       /// 3: mint_a
       /// 4: mint_b
       /// 5: user_token_a (mut)
       /// 6: user_token_b (mut)
       /// 7: swap_authority (signer)
       /// 8: user (signer)
       /// 9: token_program_a
       /// 10: token_program_b
       /// + remaining_accounts for transfer hooks
       pub fn execute_swap<'info>(
           ctx: Context<'_, '_, 'info, 'info, ExecuteSwap<'info>>,
           amount_in: u64,
           direction: u8,
           minimum_amount_out: u64,
       ) -> Result<()> {
           let bump = ctx.bumps.swap_authority;
           let signer_seeds: &[&[&[u8]]] = &[&[SWAP_AUTHORITY_SEED, &[bump]]];

           // Build AMM swap instruction discriminator
           // swap_sol_pool discriminator: sha256("global:swap_sol_pool")[0..8]
           // We'll use raw bytes - Anchor generates these at compile time
           // For now, we build instruction data manually
           let mut ix_data = Vec::with_capacity(8 + 8 + 1 + 8);
           // Discriminator for swap_sol_pool: will be filled by test setup
           // that reads the IDL discriminator
           ix_data.extend_from_slice(&ctx.accounts.instruction_discriminator.data.borrow()[..8]);
           ix_data.extend_from_slice(&amount_in.to_le_bytes());
           ix_data.extend_from_slice(&[direction]);
           ix_data.extend_from_slice(&minimum_amount_out.to_le_bytes());

           // Build account metas - must match AMM swap instruction order exactly
           let mut account_metas = vec![
               AccountMeta::new(ctx.accounts.pool.key(), false),
               AccountMeta::new(ctx.accounts.vault_a.key(), false),
               AccountMeta::new(ctx.accounts.vault_b.key(), false),
               AccountMeta::new_readonly(ctx.accounts.mint_a.key(), false),
               AccountMeta::new_readonly(ctx.accounts.mint_b.key(), false),
               AccountMeta::new(ctx.accounts.user_token_a.key(), false),
               AccountMeta::new(ctx.accounts.user_token_b.key(), false),
               AccountMeta::new_readonly(ctx.accounts.swap_authority.key(), true), // signer
               AccountMeta::new_readonly(ctx.accounts.user.key(), true), // signer
               AccountMeta::new_readonly(ctx.accounts.token_program_a.key(), false),
               AccountMeta::new_readonly(ctx.accounts.token_program_b.key(), false),
           ];

           // Append remaining accounts for transfer hooks
           for acc in ctx.remaining_accounts {
               account_metas.push(AccountMeta {
                   pubkey: acc.key(),
                   is_signer: acc.is_signer,
                   is_writable: acc.is_writable,
               });
           }

           let ix = Instruction {
               program_id: ctx.accounts.amm_program.key(),
               accounts: account_metas,
               data: ix_data,
           };

           // Collect all account infos for CPI
           let mut account_infos: Vec<AccountInfo<'info>> = vec![
               ctx.accounts.pool.to_account_info(),
               ctx.accounts.vault_a.to_account_info(),
               ctx.accounts.vault_b.to_account_info(),
               ctx.accounts.mint_a.to_account_info(),
               ctx.accounts.mint_b.to_account_info(),
               ctx.accounts.user_token_a.to_account_info(),
               ctx.accounts.user_token_b.to_account_info(),
               ctx.accounts.swap_authority.to_account_info(),
               ctx.accounts.user.to_account_info(),
               ctx.accounts.token_program_a.to_account_info(),
               ctx.accounts.token_program_b.to_account_info(),
               ctx.accounts.amm_program.to_account_info(),
           ];

           // Add remaining accounts
           for acc in ctx.remaining_accounts {
               account_infos.push(acc.clone());
           }

           invoke_signed(&ix, &account_infos, signer_seeds)?;

           Ok(())
       }
   }

   #[derive(Accounts)]
   pub struct ExecuteSwap<'info> {
       /// swap_authority PDA owned by this program.
       /// This is what we sign via invoke_signed.
       #[account(
           seeds = [SWAP_AUTHORITY_SEED],
           bump,
       )]
       pub swap_authority: SystemAccount<'info>,

       /// User executing the swap - must sign for token transfers
       pub user: Signer<'info>,

       // --- AMM accounts (passed through) ---

       /// CHECK: AMM pool account - validated by AMM program
       #[account(mut)]
       pub pool: UncheckedAccount<'info>,

       /// CHECK: Vault A - validated by AMM program
       #[account(mut)]
       pub vault_a: UncheckedAccount<'info>,

       /// CHECK: Vault B - validated by AMM program
       #[account(mut)]
       pub vault_b: UncheckedAccount<'info>,

       /// CHECK: Mint A - validated by AMM program
       pub mint_a: UncheckedAccount<'info>,

       /// CHECK: Mint B - validated by AMM program
       pub mint_b: UncheckedAccount<'info>,

       /// CHECK: User token A - validated by AMM program
       #[account(mut)]
       pub user_token_a: UncheckedAccount<'info>,

       /// CHECK: User token B - validated by AMM program
       #[account(mut)]
       pub user_token_b: UncheckedAccount<'info>,

       /// CHECK: Token program A - validated by AMM program
       pub token_program_a: UncheckedAccount<'info>,

       /// CHECK: Token program B - validated by AMM program
       pub token_program_b: UncheckedAccount<'info>,

       /// CHECK: AMM program to CPI into
       pub amm_program: UncheckedAccount<'info>,

       /// CHECK: Instruction discriminator data account
       /// Contains the 8-byte discriminator for the target AMM instruction.
       /// This is a workaround since we can't import AMM types directly.
       pub instruction_discriminator: UncheckedAccount<'info>,
   }
   ```
