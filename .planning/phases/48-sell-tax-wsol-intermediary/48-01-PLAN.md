<objective>
Modify the Tax Program's `swap_sol_sell` instruction so that sell tax is deducted from the WSOL swap output via a protocol-owned WSOL intermediary account, instead of from the user's native SOL balance. Also add an `initialize_wsol_intermediary` admin instruction that creates the intermediary PDA before the first sell. This eliminates the requirement for sellers to hold native SOL equal to the tax amount.

Purpose: This is the on-chain fix for FIX-01 (TAX-006 mainnet blocker). Users with low SOL balances can sell CRIME/FRAUD tokens. Without this, any user who has tokens but <0.05 SOL native cannot sell.

Output: Modified Rust source files for the Tax Program. After this plan, `anchor build -p tax_program` will produce an IDL with both the updated `swap_sol_sell` (new `wsol_intermediary` account in SwapSolSell) and the new `initialize_wsol_intermediary` instruction.

```rust
//! One-time initialization of the WSOL intermediary PDA account.
//! Called during protocol deployment (admin-only).

use anchor_lang::prelude::*;
use anchor_lang::solana_program::{
    instruction::{AccountMeta, Instruction},
    program::{invoke, invoke_signed},
    system_instruction,
};

use crate::constants::SWAP_AUTHORITY_SEED;
use crate::constants::WSOL_INTERMEDIARY_SEED;

pub fn handler(ctx: Context<InitializeWsolIntermediary>) -> Result<()> {
    let rent = Rent::get()?;
    let space = 165u64; // spl_token::state::Account::LEN
    let rent_lamports = rent.minimum_balance(space as usize);

    // Step 1: Create account at PDA address
    let intermediary_seeds: &[&[u8]] = &[
        WSOL_INTERMEDIARY_SEED,
        &[ctx.bumps.wsol_intermediary],
    ];

    let create_ix = system_instruction::create_account(
        ctx.accounts.admin.key,
        ctx.accounts.wsol_intermediary.key,
        rent_lamports,
        space,
        &ctx.accounts.token_program.key(),
    );

    invoke_signed(
        &create_ix,
        &[
            ctx.accounts.admin.to_account_info(),
            ctx.accounts.wsol_intermediary.to_account_info(),
            ctx.accounts.system_program.to_account_info(),
        ],
        &[intermediary_seeds],
    )?;

    // Step 2: Initialize as WSOL token account using InitializeAccount3
    // InitializeAccount3 (discriminator 18) does NOT require rent sysvar.
    // Owner is passed as instruction data (32 bytes after discriminator).
    let init_ix = Instruction {
        program_id: ctx.accounts.token_program.key(),
        accounts: vec![
            AccountMeta::new(ctx.accounts.wsol_intermediary.key(), false),
            AccountMeta::new_readonly(ctx.accounts.mint.key(), false),
        ],
        data: {
            let mut d = vec![18u8]; // InitializeAccount3 discriminator
            d.extend_from_slice(&ctx.accounts.swap_authority.key().to_bytes());
            d
        },
    };

    invoke(
        &init_ix,
        &[
            ctx.accounts.wsol_intermediary.to_account_info(),
            ctx.accounts.mint.to_account_info(),
            ctx.accounts.token_program.to_account_info(),
        ],
    )?;

    msg!("WSOL intermediary initialized at {}", ctx.accounts.wsol_intermediary.key());
    Ok(())
}

#[derive(Accounts)]
pub struct InitializeWsolIntermediary<'info> {
    /// Admin (payer). Only needs to be called once.
    #[account(mut)]
    pub admin: Signer<'info>,

    /// WSOL intermediary PDA -- must not exist yet.
    /// CHECK: PDA derived from seeds. Will be created in handler.
    #[account(
        mut,
        seeds = [WSOL_INTERMEDIARY_SEED],
        bump,
```
