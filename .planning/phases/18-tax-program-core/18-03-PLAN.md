    Implement the handler function following Tax_Pool_Logic_Spec.md Section 14.2 (BUY: SOL -> IP):

    ```rust
    pub fn handler<'info>(
        ctx: Context<'_, '_, 'info, 'info, SwapSolBuy<'info>>,
        amount_in: u64,
        minimum_output: u64,
        is_crime: bool,
    ) -> Result<()> {
        // 1. Read tax rate from epoch_state based on is_crime
        //    For now, mock with hardcoded rate or read from account data
        let tax_bps = if is_crime {
            // epoch_state.crime_buy_tax_bps
            400u16  // TODO: Read from EpochState when available
        } else {
            // epoch_state.fraud_buy_tax_bps
            400u16  // TODO: Read from EpochState when available
        };

        // 2. Calculate tax: tax = amount_in * tax_bps / 10_000
        let tax_amount = crate::helpers::tax_math::calculate_tax(amount_in, tax_bps)
            .ok_or(error!(TaxError::TaxOverflow))?;

        // 3. Calculate sol_to_swap = amount_in - tax
        let sol_to_swap = amount_in
            .checked_sub(tax_amount)
            .ok_or(error!(TaxError::TaxOverflow))?;

        // 4. Split distribution: 75% staking, 24% carnage, 1% treasury
        let (staking_portion, carnage_portion, treasury_portion) =
            crate::helpers::tax_math::split_distribution(tax_amount)
                .ok_or(error!(TaxError::TaxOverflow))?;

        // 5. Execute native SOL transfers for tax distribution
        //    Transfer from user to staking_escrow
        if staking_portion > 0 {
            solana_program::program::invoke(
                &solana_program::system_instruction::transfer(
                    ctx.accounts.user.key,
                    ctx.accounts.staking_escrow.key,
                    staking_portion,
                ),
                &[
                    ctx.accounts.user.to_account_info(),
                    ctx.accounts.staking_escrow.to_account_info(),
                    ctx.accounts.system_program.to_account_info(),
                ],
            )?;
        }
        //    Transfer from user to carnage_vault
        if carnage_portion > 0 {
            solana_program::program::invoke(
                &solana_program::system_instruction::transfer(
                    ctx.accounts.user.key,
                    ctx.accounts.carnage_vault.key,
                    carnage_portion,
                ),
                &[
                    ctx.accounts.user.to_account_info(),
                    ctx.accounts.carnage_vault.to_account_info(),
                    ctx.accounts.system_program.to_account_info(),
                ],
            )?;
        }
        //    Transfer from user to treasury
        if treasury_portion > 0 {
            solana_program::program::invoke(
                &solana_program::system_instruction::transfer(
                    ctx.accounts.user.key,
                    ctx.accounts.treasury.key,
                    treasury_portion,
                ),
                &[
                    ctx.accounts.user.to_account_info(),
                    ctx.accounts.treasury.to_account_info(),
                    ctx.accounts.system_program.to_account_info(),
                ],
            )?;
        }

        // 6. Build AMM CPI instruction
        //    Use invoke_signed with swap_authority PDA
        //    Build instruction data for AMM::swap_sol_pool(sol_to_swap, AtoB, minimum_output)
        //    Direction AtoB = SOL (A) -> Token (B)
        let swap_authority_seeds: &[&[u8]] = &[
            b"swap_authority",
            &[ctx.bumps.swap_authority],
        ];

        // Build account metas for AMM CPI (same pattern as mock-tax-program)
        // AMM accounts: swap_authority, pool, vault_a, vault_b, mint_a, mint_b,
        //               user_token_a, user_token_b, user, token_program_a, token_program_b
        let mut account_metas = vec![
            AccountMeta::new_readonly(ctx.accounts.swap_authority.key(), true),
            AccountMeta::new(ctx.accounts.pool.key(), false),
            AccountMeta::new(ctx.accounts.vault_sol.key(), false),
            AccountMeta::new(ctx.accounts.vault_token.key(), false),
            AccountMeta::new_readonly(ctx.accounts.mint_sol.key(), false),
            AccountMeta::new_readonly(ctx.accounts.mint_token.key(), false),
            AccountMeta::new(ctx.accounts.user_wsol.key(), false),
            AccountMeta::new(ctx.accounts.user_token.key(), false),
            AccountMeta::new_readonly(ctx.accounts.user.key(), true),
            AccountMeta::new_readonly(ctx.accounts.token_program_sol.key(), false),
            AccountMeta::new_readonly(ctx.accounts.token_program_token.key(), false),
        ];

        // Add remaining_accounts for transfer hook
        for account in ctx.remaining_accounts.iter() {
            if account.is_writable {
                account_metas.push(AccountMeta::new(account.key(), account.is_signer));
            } else {
                account_metas.push(AccountMeta::new_readonly(account.key(), account.is_signer));
            }
        }

        // Build instruction data: discriminator + args
        // AMM swap_sol_pool discriminator + (sol_to_swap, direction=0, minimum_output)
        let amm_discriminator = anchor_lang::solana_program::hash::hash(b"global:swap_sol_pool").to_bytes()[..8].to_vec();
        let mut ix_data = amm_discriminator;
        ix_data.extend_from_slice(&sol_to_swap.to_le_bytes());
        ix_data.push(0u8);  // SwapDirection::AtoB = 0
        ix_data.extend_from_slice(&minimum_output.to_le_bytes());

        let ix = solana_program::instruction::Instruction {
            program_id: ctx.accounts.amm_program.key(),
            accounts: account_metas,
            data: ix_data,
        };

        // Build account infos for CPI
        let mut account_infos = vec![
            ctx.accounts.swap_authority.to_account_info(),
            ctx.accounts.pool.to_account_info(),
            ctx.accounts.vault_sol.to_account_info(),
            ctx.accounts.vault_token.to_account_info(),
            ctx.accounts.mint_sol.to_account_info(),
            ctx.accounts.mint_token.to_account_info(),
            ctx.accounts.user_wsol.to_account_info(),
            ctx.accounts.user_token.to_account_info(),
            ctx.accounts.user.to_account_info(),
            ctx.accounts.token_program_sol.to_account_info(),
            ctx.accounts.token_program_token.to_account_info(),
        ];
        account_infos.extend(ctx.remaining_accounts.iter().cloned());
        account_infos.push(ctx.accounts.amm_program.to_account_info());

        // Execute CPI
        solana_program::program::invoke_signed(
            &ix,
            &account_infos,
            &[swap_authority_seeds],
        )?;

        // 7. Emit TaxedSwap event
        let clock = Clock::get()?;
        emit!(TaxedSwap {
            user: ctx.accounts.user.key(),
            pool_type: if is_crime { PoolType::SolCrime } else { PoolType::SolFraud },
            direction: SwapDirection::Buy,
            input_amount: amount_in,
            output_amount: 0, // TODO: Get from CPI return or compute
            tax_amount,
            tax_rate_bps: tax_bps,
            staking_portion,
            carnage_portion,
            treasury_portion,
            epoch: 0,  // TODO: Read from EpochState
            slot: clock.slot,
        });

        Ok(())
    }
    ```
