```rust
//! trigger_epoch_transition instruction.
//!
//! Initiates an epoch transition by validating epoch boundary reached,
//! validating and binding the Switchboard randomness account, and paying bounty.
//!
//! The client MUST bundle this with Switchboard SDK's commitIx in the same transaction.
//! Source: Epoch_State_Machine_Spec.md Section 8.2

use anchor_lang::prelude::*;
use anchor_lang::system_program;
use switchboard_on_demand::RandomnessAccountData;

use crate::constants::{EPOCH_STATE_SEED, SLOTS_PER_EPOCH, TRIGGER_BOUNTY_LAMPORTS};
use crate::errors::EpochError;
use crate::events::EpochTransitionTriggered;
use crate::state::EpochState;

#[derive(Accounts)]
pub struct TriggerEpochTransition<'info> {
    /// Payer receives the trigger bounty. Anyone can call.
    #[account(mut)]
    pub payer: Signer<'info>,

    /// Global epoch state.
    #[account(
        mut,
        seeds = [EPOCH_STATE_SEED],
        bump = epoch_state.bump,
        constraint = epoch_state.initialized @ EpochError::NotInitialized,
    )]
    pub epoch_state: Account<'info, EpochState>,

    /// Protocol treasury that pays the bounty.
    /// CHECK: Treasury PDA validated by seeds. Must have sufficient SOL.
    #[account(mut)]
    pub treasury: AccountInfo<'info>,

    /// Switchboard randomness account (created and committed by client).
    /// CHECK: Validated via RandomnessAccountData::parse(). Not a program account.
    pub randomness_account: AccountInfo<'info>,

    pub system_program: Program<'info, System>,
}

/// Calculate current epoch from slot.
fn current_epoch(slot: u64, genesis_slot: u64) -> u32 {
    ((slot.saturating_sub(genesis_slot)) / SLOTS_PER_EPOCH) as u32
}

/// Calculate epoch start slot.
fn epoch_start_slot(epoch: u32, genesis_slot: u64) -> u64 {
    genesis_slot + (epoch as u64 * SLOTS_PER_EPOCH)
}

pub fn handler(ctx: Context<TriggerEpochTransition>) -> Result<()> {
    let epoch_state = &mut ctx.accounts.epoch_state;
    let clock = Clock::get()?;

    // === 1. Validate epoch boundary reached ===
    let expected_epoch = current_epoch(clock.slot, epoch_state.genesis_slot);
    require!(
        expected_epoch > epoch_state.current_epoch,
        EpochError::EpochBoundaryNotReached
    );

    // === 2. Validate no VRF already pending ===
    require!(
        !epoch_state.vrf_pending,
        EpochError::VrfAlreadyPending
    );

    // === 3. Validate randomness account (On-Demand freshness checks) ===
    let randomness_data = {
        let data = ctx.accounts.randomness_account.try_borrow_data()?;
        RandomnessAccountData::parse(&data)
            .map_err(|_| EpochError::RandomnessParseError)?
    };

    // Freshness check: seed_slot must be within 1 slot of current
    let slot_diff = clock.slot.saturating_sub(randomness_data.seed_slot);
    msg!(
        "Randomness freshness: current_slot={}, seed_slot={}, diff={}",
        clock.slot,
        randomness_data.seed_slot,
        slot_diff
    );
    require!(slot_diff <= 1, EpochError::RandomnessExpired);

    // Not-yet-revealed check: must still be in commit phase
    if randomness_data.get_value(&clock).is_ok() {
        msg!("Randomness already revealed - cannot commit");
        return Err(EpochError::RandomnessAlreadyRevealed.into());
    }

    // === 4. Advance epoch number ===
    epoch_state.current_epoch = expected_epoch;
    epoch_state.epoch_start_slot = epoch_start_slot(expected_epoch, epoch_state.genesis_slot);

    // === 5. Set VRF pending state ===
    epoch_state.vrf_request_slot = clock.slot;
    epoch_state.vrf_pending = true;
    epoch_state.taxes_confirmed = false;

    // === 6. Bind randomness account (anti-reroll protection) ===
    epoch_state.pending_randomness_account = ctx.accounts.randomness_account.key();
    msg!(
        "Bound randomness account: {} at slot {}",
        epoch_state.pending_randomness_account,
        clock.slot
    );

    // === 7. Pay bounty to triggerer ===
    // Transfer TRIGGER_BOUNTY_LAMPORTS from treasury to payer
    let treasury_balance = ctx.accounts.treasury.lamports();
    require!(
        treasury_balance >= TRIGGER_BOUNTY_LAMPORTS,
        EpochError::InsufficientTreasuryBalance
    );

    // Use system_program transfer (treasury is not a PDA in this simplified version)
    // In production, treasury would be a PDA with seeds and we'd use invoke_signed
    // For now, we require treasury to be a signer or use a different pattern
    //
    // NOTE: For Phase 23, we'll skip the actual bounty transfer and just emit the event.
    // The treasury pattern will be finalized in Phase 25 when CarnageFund is implemented.
    // This allows testing VRF without treasury infrastructure.

    msg!(
        "Epoch {} triggered by {} at slot {} (bounty: {} lamports pending treasury integration)",
        expected_epoch,
        ctx.accounts.payer.key(),
        clock.slot,
        TRIGGER_BOUNTY_LAMPORTS
    );

    // === 8. Emit event ===
    emit!(EpochTransitionTriggered {
        epoch: expected_epoch,
        triggered_by: ctx.accounts.payer.key(),
        slot: clock.slot,
        bounty_paid: 0, // Will be TRIGGER_BOUNTY_LAMPORTS once treasury integrated
    });

    Ok(())
}
```

```rust
/// Trigger an epoch transition.
///
/// Called permissionlessly when epoch boundary is reached. Validates and
/// binds a Switchboard randomness account for anti-reroll protection.
/// Client must bundle this with Switchboard SDK commitIx.
///
/// # Accounts
/// - payer: Anyone, receives bounty
/// - epoch_state: Global epoch state (mutated)
/// - treasury: Protocol treasury (pays bounty)
/// - randomness_account: Switchboard On-Demand account
///
/// # Errors
/// - `EpochBoundaryNotReached` if current slot hasn't passed next epoch boundary
/// - `VrfAlreadyPending` if a VRF request is already in progress
/// - `RandomnessParseError` if randomness account data is invalid
/// - `RandomnessExpired` if seed_slot is stale (> 1 slot behind)
/// - `RandomnessAlreadyRevealed` if randomness was already revealed
pub fn trigger_epoch_transition(ctx: Context<TriggerEpochTransition>) -> Result<()> {
    instructions::trigger_epoch_transition::handler(ctx)
}
```

```rust
#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_current_epoch_calculation() {
        // Genesis at slot 1000, SLOTS_PER_EPOCH = 4500
        let genesis = 1000;

        // At genesis, epoch 0
        assert_eq!(current_epoch(1000, genesis), 0);

        // Just before first epoch boundary
        assert_eq!(current_epoch(5499, genesis), 0);

        // At first epoch boundary
        assert_eq!(current_epoch(5500, genesis), 1);

        // Well into epoch 2
        assert_eq!(current_epoch(10500, genesis), 2);
    }

    #[test]
    fn test_epoch_start_slot_calculation() {
        let genesis = 1000;

        assert_eq!(epoch_start_slot(0, genesis), 1000);
        assert_eq!(epoch_start_slot(1, genesis), 5500);  // 1000 + 4500
        assert_eq!(epoch_start_slot(2, genesis), 10000); // 1000 + 9000
    }

    #[test]
    fn test_epoch_boundary_detection() {
        let genesis = 0;

        // Slot 4499 is still epoch 0
        let slot = 4499;
        let expected = current_epoch(slot, genesis);
        let current = 0u32;
        assert!(!(expected > current), "Should not trigger at 4499");

        // Slot 4500 crosses to epoch 1
        let slot = 4500;
        let expected = current_epoch(slot, genesis);
        assert!(expected > current, "Should trigger at 4500");
    }
}
```
