//! Tax rate derivation from VRF randomness.
//!
//! Converts VRF bytes to discrete tax bands per Epoch_State_Machine_Spec.md Section 7.3.

use crate::state::Token;

/// Low tax band rates in basis points (1%, 2%, 3%, 4%).
/// VRF byte % 4 selects index.
pub const LOW_RATES: [u16; 4] = [100, 200, 300, 400];

/// High tax band rates in basis points (11%, 12%, 13%, 14%).
/// VRF byte % 4 selects index.
pub const HIGH_RATES: [u16; 4] = [1100, 1200, 1300, 1400];

/// Threshold for 75% flip probability.
/// VRF byte 0 < 192 triggers flip (192/256 = 75%).
pub const FLIP_THRESHOLD: u8 = 192;

/// Tax configuration derived from VRF randomness.
#[derive(Clone, Copy, Debug, PartialEq)]
pub struct TaxConfig {
    pub cheap_side: Token,
    pub low_tax_bps: u16,
    pub high_tax_bps: u16,
    pub crime_buy_tax_bps: u16,
    pub crime_sell_tax_bps: u16,
    pub fraud_buy_tax_bps: u16,
    pub fraud_sell_tax_bps: u16,
}

/// Derive tax configuration from VRF randomness bytes.
///
/// VRF byte allocation (per Epoch_State_Machine_Spec.md Section 7.2):
/// - Byte 0: Regime flip (< 192 = 75% flip probability)
/// - Byte 1: Low tax magnitude (% 4 selects from LOW_RATES)
/// - Byte 2: High tax magnitude (% 4 selects from HIGH_RATES)
///
/// # Arguments
/// * `vrf_bytes` - At least 3 bytes of VRF randomness
/// * `current_cheap` - Current cheap side before this epoch
///
/// # Returns
/// TaxConfig with new cheap side and all four derived tax rates
pub fn derive_taxes(vrf_bytes: &[u8], current_cheap: Token) -> TaxConfig {
    // Byte 0: Flip decision (75% probability)
    let should_flip = vrf_bytes[0] < FLIP_THRESHOLD;

    // Byte 1: Low tax magnitude (4 discrete values)
    let low_bps = LOW_RATES[(vrf_bytes[1] % 4) as usize];

    // Byte 2: High tax magnitude (4 discrete values)
    let high_bps = HIGH_RATES[(vrf_bytes[2] % 4) as usize];

    // Determine new cheap side
    let cheap_side = if should_flip {
        current_cheap.opposite()
    } else {
        current_cheap
    };

    // Derive all four rates based on cheap side
    // When CRIME is cheap: CRIME buy=low, CRIME sell=high, FRAUD buy=high, FRAUD sell=low
    // When FRAUD is cheap: CRIME buy=high, CRIME sell=low, FRAUD buy=low, FRAUD sell=high
    let (crime_buy, crime_sell, fraud_buy, fraud_sell) = match cheap_side {
        Token::Crime => (low_bps, high_bps, high_bps, low_bps),
        Token::Fraud => (high_bps, low_bps, low_bps, high_bps),
    };

    TaxConfig {
        cheap_side,
        low_tax_bps: low_bps,
        high_tax_bps: high_bps,
        crime_buy_tax_bps: crime_buy,
        crime_sell_tax_bps: crime_sell,
        fraud_buy_tax_bps: fraud_buy,
        fraud_sell_tax_bps: fraud_sell,
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_flip_threshold_75_percent() {
        // 192/256 = 0.75 = 75%
        assert!(191 < FLIP_THRESHOLD); // Should flip
        assert!(192 >= FLIP_THRESHOLD); // Should not flip
    }

    #[test]
    fn test_low_rates_selection() {
        // byte % 4 maps to index
        assert_eq!(LOW_RATES[0], 100); // 1%
        assert_eq!(LOW_RATES[1], 200); // 2%
        assert_eq!(LOW_RATES[2], 300); // 3%
        assert_eq!(LOW_RATES[3], 400); // 4%
    }

    #[test]
    fn test_high_rates_selection() {
        assert_eq!(HIGH_RATES[0], 1100); // 11%
        assert_eq!(HIGH_RATES[1], 1200); // 12%
        assert_eq!(HIGH_RATES[2], 1300); // 13%
        assert_eq!(HIGH_RATES[3], 1400); // 14%
    }

    #[test]
    fn test_derive_taxes_no_flip() {
        // VRF byte 0 = 200 (>= 192, no flip)
        // VRF byte 1 = 5 (5 % 4 = 1, low_bps = 200)
        // VRF byte 2 = 10 (10 % 4 = 2, high_bps = 1300)
        let vrf = [200, 5, 10];
        let config = derive_taxes(&vrf, Token::Crime);

        assert_eq!(config.cheap_side, Token::Crime); // No flip
        assert_eq!(config.low_tax_bps, 200);
        assert_eq!(config.high_tax_bps, 1300);
        assert_eq!(config.crime_buy_tax_bps, 200); // Cheap = low
        assert_eq!(config.crime_sell_tax_bps, 1300); // Cheap sell = high
        assert_eq!(config.fraud_buy_tax_bps, 1300); // Expensive = high
        assert_eq!(config.fraud_sell_tax_bps, 200); // Expensive sell = low
    }

    #[test]
    fn test_derive_taxes_with_flip() {
        // VRF byte 0 = 100 (< 192, flip!)
        // VRF byte 1 = 0 (0 % 4 = 0, low_bps = 100)
        // VRF byte 2 = 3 (3 % 4 = 3, high_bps = 1400)
        let vrf = [100, 0, 3];
        let config = derive_taxes(&vrf, Token::Crime);

        assert_eq!(config.cheap_side, Token::Fraud); // Flipped from Crime
        assert_eq!(config.low_tax_bps, 100);
        assert_eq!(config.high_tax_bps, 1400);
        assert_eq!(config.fraud_buy_tax_bps, 100); // FRAUD now cheap
        assert_eq!(config.fraud_sell_tax_bps, 1400);
        assert_eq!(config.crime_buy_tax_bps, 1400); // CRIME now expensive
        assert_eq!(config.crime_sell_tax_bps, 100);
    }

    #[test]
    fn test_all_vrf_byte_boundaries() {
        // Test boundary values for byte 1 (low rate selection)
        for i in 0u8..=255 {
            let expected_idx = (i % 4) as usize;
            let expected_rate = LOW_RATES[expected_idx];
            let vrf = [200, i, 0]; // No flip, test byte 1
            let config = derive_taxes(&vrf, Token::Crime);
            assert_eq!(config.low_tax_bps, expected_rate, "byte {} failed", i);
        }
    }
}
