    1. Replace the hardcoded `tokenMetadata` record with a function that resolves URIs:
       ```
       function resolveMetadataUri(tokenName: string, cluster: string): string {
         // Priority: env var > deployment.json > hard error
         const envKey = `${tokenName.toUpperCase()}_METADATA_URI`;
         const envVal = process.env[envKey];
         if (envVal) return envVal;

         // Try deployment.json
         const deployPath = path.join(__dirname, '../../deployments', `${cluster}.json`);
         if (fs.existsSync(deployPath)) {
           const config = JSON.parse(fs.readFileSync(deployPath, 'utf-8'));
           const key = tokenName.toLowerCase() as 'crime' | 'fraud' | 'profit';
           if (config.metadata?.[key]) return config.metadata[key];
         }

         throw new Error(`No metadata URI for ${tokenName}. Set ${envKey} env var or run upload-metadata.ts first.`);
       }
       ```

    2. Update the tokenMetadata record to call resolveMetadataUri for each token:

       const tokenMetadata: Record<string, { symbol: string; uri: string }> = {

         FRAUD: { symbol: "FRAUD", uri: resolveMetadataUri("FRAUD", cluster) },
         PROFIT: { symbol: "PROFIT", uri: resolveMetadataUri("PROFIT", cluster) },
