   ```rust
   #[error_code]
   pub enum StakingError {
       #[msg("Amount must be greater than zero")]
       ZeroAmount,  // ERR-01

       #[msg("Insufficient staked balance")]
       InsufficientBalance,  // ERR-02

       #[msg("Insufficient SOL in escrow vault")]
       InsufficientEscrowBalance,  // ERR-03

       #[msg("No rewards to claim")]
       NothingToClaim,  // ERR-04

       #[msg("Unauthorized: signer does not own this stake account")]
       Unauthorized,  // ERR-05

       #[msg("Arithmetic overflow")]
       Overflow,

       #[msg("Arithmetic underflow")]
       Underflow,

       #[msg("Division by zero")]
       DivisionByZero,

       #[msg("Cumulative already updated for this epoch")]
       AlreadyUpdated,

       #[msg("Pool not initialized")]
       NotInitialized,

       #[msg("Pool already initialized")]
       AlreadyInitialized,
   }
   ```
