Create `parseStakingError(error: unknown): string` following the same pattern as `parseSwapError`:

**Step 2: Create `app/lib/staking/rewards.ts`**

This file contains the client-side reward math that mirrors the on-chain `update_rewards()` function from `programs/staking/src/helpers/math.rs`.

**Export 1: `calculatePendingRewards()`**

```typescript
const PRECISION = BigInt("1000000000000000000"); // 1e18, matches on-chain

export function calculatePendingRewards(
  poolRewardsPerTokenStored: bigint,  // from stakePool.rewardsPerTokenStored (u128)
  userRewardsPerTokenPaid: bigint,    // from userStake.rewardsPerTokenPaid (u128)
  userStakedBalance: bigint,          // from userStake.stakedBalance (u64)
  userRewardsEarned: bigint,          // from userStake.rewardsEarned (u64)
): number {
  const delta = poolRewardsPerTokenStored - userRewardsPerTokenPaid;
  const newPending = (userStakedBalance * delta) / PRECISION;
  const totalPending = userRewardsEarned + newPending;
  return Number(totalPending); // Safe: max SOL supply is ~5e17 lamports
}
```

CRITICAL: This function takes `bigint` parameters. The hook (Plan 02) will convert Anchor BN values to BigInt before calling this function. Example conversion: `BigInt(bn.toString())`.

**Export 2: `calculateRewardRate()`**

```typescript
export interface RewardRateStats {
  perEpochLamports: number;  // SOL per epoch (from pendingRewards)
  annualizedPct: number;     // Annualized percentage
  totalStakedProfit: number; // Total PROFIT staked (display units)
  userSharePct: number;      // User's share of pool (0-100)
}

/**
 * Calculate reward rate statistics.
 *
 * For v1, uses pending_rewards as proxy for "this epoch's rewards".
 * This is accurate between update_cumulative calls but resets to 0
 * immediately after finalization.
 *
 * Annualized calculation:
 * - Epoch length: ~100 slots = ~40 seconds (devnet)
 * - Epochs per year: 365.25 * 24 * 3600 / 40 = ~788,940
 * - Per-staker per-epoch: (pendingRewards * userShare) / totalStaked
 * - Annualized: perEpochReward * epochsPerYear / userStake * 100
 */
const EPOCH_DURATION_SECONDS = 40; // ~100 slots * 0.4s/slot on devnet
const SECONDS_PER_YEAR = 365.25 * 24 * 3600;
const EPOCHS_PER_YEAR = SECONDS_PER_YEAR / EPOCH_DURATION_SECONDS;
const PROFIT_DECIMALS = 6;
const LAMPORTS_PER_SOL = 1_000_000_000;

export function calculateRewardRate(
  pendingRewardsLamports: number,
  totalStakedBaseUnits: number,
  userStakedBaseUnits: number,
  solPricePerProfit?: number, // Optional: SOL per PROFIT for APY calc
): RewardRateStats {
  const totalStakedProfit = totalStakedBaseUnits / 10 ** PROFIT_DECIMALS;
  const userSharePct = totalStakedBaseUnits > 0
    ? (userStakedBaseUnits / totalStakedBaseUnits) * 100
    : 0;

  // Annualized: if total_staked > 0 and we have SOL/PROFIT price
  // APR = (reward_per_epoch * epochs_per_year) / (total_staked_in_SOL_terms) * 100
  // For simplicity (v1): show raw per-epoch SOL amount, and if we have a price ratio,
  // compute the annualized percentage.
  let annualizedPct = 0;
  if (totalStakedBaseUnits > 0 && pendingRewardsLamports > 0 && solPricePerProfit) {
    const totalStakedInSol = totalStakedBaseUnits * solPricePerProfit / 10 ** PROFIT_DECIMALS;
    const annualRewardsInSol = (pendingRewardsLamports / LAMPORTS_PER_SOL) * EPOCHS_PER_YEAR;
    annualizedPct = (annualRewardsInSol / totalStakedInSol) * 100;
  }

  return {
    perEpochLamports: pendingRewardsLamports,
    annualizedPct,
    totalStakedProfit,
    userSharePct,
  };
}
```

Also export PRECISION, PROFIT_DECIMALS, and LAMPORTS_PER_SOL constants since the hook may need them.

Note: The annualized percentage requires knowing the SOL value of staked PROFIT, which requires pool price data. The hook will pass this from usePoolPrices. If not available, annualizedPct defaults to 0 and the UI should hide the annualized line.
