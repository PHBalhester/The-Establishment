1. **assertEscrowSolvency helper**:
```typescript
async function assertEscrowSolvency(
  connection: Connection,
  stakingProgram: Program<Staking>,
  escrowVault: PublicKey,
  stakePool: PublicKey,
): Promise<void> {
  const pool = await stakingProgram.account.stakePool.fetch(stakePool);
  const escrowBalance = await connection.getBalance(escrowVault);

  // Escrow must cover pending rewards
  // Note: pendingRewards is the undistributed amount before cumulative update
  expect(escrowBalance).to.be.gte(
    pool.pendingRewards.toNumber(),
    `Escrow solvency violated: balance ${escrowBalance} < pending ${pool.pendingRewards.toNumber()}`
  );
}
```

2. **Happy path test suite**:
```typescript
describe("happy path", () => {
  it("stake transfers PROFIT through Transfer Hook", async () => {
    const amount = 1_000_000_000; // 1000 PROFIT

    const beforeVault = await getAccount(connection, stakeVault, undefined, TOKEN_2022_PROGRAM_ID);

    await stakeWithHook(connection, stakingProgram, user, amount, {
      stakePool, userStake, userTokenAccount, stakeVault, profitMint
    });

    const afterVault = await getAccount(connection, stakeVault, undefined, TOKEN_2022_PROGRAM_ID);
    expect(Number(afterVault.amount)).to.equal(Number(beforeVault.amount) + amount);

    await assertEscrowSolvency(connection, stakingProgram, escrowVault, stakePool);
  });

  it("unstake transfers PROFIT back through Transfer Hook", async () => {
    const amount = 500_000_000; // 500 PROFIT

    const beforeUser = await getAccount(connection, userTokenAccount, undefined, TOKEN_2022_PROGRAM_ID);

    await unstakeWithHook(connection, stakingProgram, user, amount, {
      stakePool, userStake, userTokenAccount, stakeVault, escrowVault, profitMint
    });

    const afterUser = await getAccount(connection, userTokenAccount, undefined, TOKEN_2022_PROGRAM_ID);
    expect(Number(afterUser.amount)).to.equal(Number(beforeUser.amount) + amount);

    await assertEscrowSolvency(connection, stakingProgram, escrowVault, stakePool);
  });
});
```
