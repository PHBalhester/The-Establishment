```rust
#[test]
fn test_profit_pool_produces_more_output_than_sol_pool()
// Given identical reserves and input amount:
// - PROFIT pool (50 bps) should produce MORE output than SOL pool (100 bps)
//
// This test verifies the fee difference is reflected in outputs.
//
// Setup:
// 1. Create two pools with identical reserves (can use math helpers, no need for on-chain)
// 2. Calculate expected outputs for same input using both fee rates
// 3. Assert profit_output > sol_output
//
// Implementation:
let amount_in = 100_000_000u64; // 0.1 token
let reserve = 1_000_000_000u64; // 1 token each side

let sol_effective = expected_effective_input(amount_in, SOL_POOL_FEE_BPS);
let profit_effective = expected_effective_input(amount_in, PROFIT_POOL_FEE_BPS);

let sol_output = expected_swap_output(reserve, reserve, sol_effective);
let profit_output = expected_swap_output(reserve, reserve, profit_effective);

assert!(profit_output > sol_output,
    "PROFIT pool (50 bps) should produce more output than SOL pool (100 bps): \
     profit_output={}, sol_output={}", profit_output, sol_output);

// Also verify the magnitude of difference is reasonable
// With 50 bps vs 100 bps, profit_effective = 99.5% of input, sol_effective = 99% of input
// So profit_output should be slightly more (not dramatically more)
let output_ratio = (profit_output as f64) / (sol_output as f64);
assert!(output_ratio > 1.0 && output_ratio < 1.02,
    "Output ratio should be between 1.0 and 1.02 (slight improvement from lower fee)");

#[test]
fn test_fee_calculation_consistency()
// Verify fee amounts scale correctly with input
// For PROFIT pool: fee = amount_in * 50 / 10000 = amount_in * 0.005
// For SOL pool: fee = amount_in * 100 / 10000 = amount_in * 0.01
// PROFIT fee should be exactly half of SOL fee

let amount_in = 100_000_000u64;

let profit_effective = expected_effective_input(amount_in, PROFIT_POOL_FEE_BPS);
let sol_effective = expected_effective_input(amount_in, SOL_POOL_FEE_BPS);

let profit_fee = amount_in - (profit_effective as u64);
let sol_fee = amount_in - (sol_effective as u64);

assert_eq!(profit_fee * 2, sol_fee,
    "PROFIT pool fee should be exactly half of SOL pool fee: \
     profit_fee={}, sol_fee={}", profit_fee, sol_fee);
```
