/// Load compiled program from target directory
fn load_program_elf(name: &str) -> Vec<u8> {
    let manifest = std::env::var("CARGO_MANIFEST_DIR").unwrap();
    let program_path = Path::new(&manifest)
        .parent().unwrap()  // programs/
        .parent().unwrap()  // root
        .join("target/deploy")
        .join(format!("{}.so", name));

    std::fs::read(&program_path)
        .unwrap_or_else(|_| panic!("Failed to load program: {:?}", program_path))
}

/// Create a Token-2022 mint with transfer hook extension
fn create_t22_mint_with_hook(
    svm: &mut LiteSVM,
    mint_keypair: &LiteKeypair,
    mint_authority: &Pubkey,
    decimals: u8,
    hook_program_id: &Pubkey,
) -> Address {
    let mint_addr = Address::from(mint_keypair.pubkey());

    // Calculate space for mint with transfer hook extension
    let extension_types = vec![ExtensionType::TransferHook];
    let space = ExtensionType::try_calculate_account_len::<T22MintState>(&extension_types).unwrap();

    // Create the mint account data
    let mut mint_data = vec![0u8; space];

    // Initialize the mint state with extensions
    let mut state = StateWithExtensionsMut::<T22MintState>::unpack_uninitialized(&mut mint_data).unwrap();

    // Initialize the extension first
    state.init_extension::<hook_ext::TransferHook>(true).unwrap();

    // Set the transfer hook program ID
    let hook_ext = state.get_extension_mut::<hook_ext::TransferHook>().unwrap();
    hook_ext.authority = spl_pod::optional_keys::OptionalNonZeroPubkey::try_from(Some(*mint_authority)).unwrap();
    hook_ext.program_id = spl_pod::optional_keys::OptionalNonZeroPubkey::try_from(Some(*hook_program_id)).unwrap();

    // Now initialize the base mint state
    state.base.mint_authority = spl_pod::primitives::PodCOption::some(*mint_authority);
    state.base.supply = 0u64.into();
    state.base.decimals = decimals;
    state.base.is_initialized = true.into();
    state.base.freeze_authority = spl_pod::primitives::PodCOption::none();

    // Pack the state back
    state.pack_base();

    let rent = svm.minimum_balance_for_rent_exemption(space);

    svm.set_account(
        mint_addr.clone(),
        Account {
            lamports: rent,
            data: mint_data,
            owner: pk_to_addr(&token_2022_program_id()).0.into(),
            executable: false,
            rent_epoch: 0,
        },
    ).unwrap();

    mint_addr
}

/// Create a Token-2022 token account with transfer hook extension
fn create_t22_token_account(
    svm: &mut LiteSVM,
    account_keypair: &LiteKeypair,
    mint: &Pubkey,
    owner: &Pubkey,
    amount: u64,
) -> Address {
    use spl_token_2022::state::Account as T22AccountState;
    use spl_token_2022::extension::transfer_hook::TransferHookAccount;

    let account_addr = Address::from(account_keypair.pubkey());

    // Calculate space for token account with transfer hook extension
    let extension_types = vec![ExtensionType::TransferHookAccount];
    let space = ExtensionType::try_calculate_account_len::<T22AccountState>(&extension_types).unwrap();

    let mut account_data = vec![0u8; space];

    // Initialize account state with extensions
    let mut state = StateWithExtensionsMut::<T22AccountState>::unpack_uninitialized(&mut account_data).unwrap();

    // Initialize the extension
    state.init_extension::<TransferHookAccount>(true).unwrap();

    // Initialize base account state
    state.base.mint = *mint;
    state.base.owner = *owner;
    state.base.amount = amount.into();
    state.base.delegate = spl_pod::primitives::PodCOption::none();
    state.base.state = spl_token_2022::state::AccountState::Initialized.into();
    state.base.is_native = spl_pod::primitives::PodCOption::none();
    state.base.delegated_amount = 0u64.into();
    state.base.close_authority = spl_pod::primitives::PodCOption::none();

    state.pack_base();

    let rent = svm.minimum_balance_for_rent_exemption(space);

    svm.set_account(
        account_addr.clone(),
        Account {
            lamports: rent,
            data: account_data,
            owner: pk_to_addr(&token_2022_program_id()).0.into(),
            executable: false,
            rent_epoch: 0,
        },
    ).unwrap();

    account_addr
}

/// Create whitelist authority PDA
fn create_whitelist_authority(
    svm: &mut LiteSVM,
    authority: &Pubkey,
) -> (Address, u8) {
    let (pda, bump) = Pubkey::find_program_address(
        &[WHITELIST_AUTHORITY_SEED],
        &program_id()
    );

    // WhitelistAuthority: discriminator (8) + Option<Pubkey> (1 + 32) = 41 bytes
    let mut data = vec![0u8; 8 + 1 + 32];

    // Anchor discriminator for WhitelistAuthority
    let mut hasher = Sha256::new();
    hasher.update(b"account:WhitelistAuthority");
    let hash = hasher.finalize();
    data[0..8].copy_from_slice(&hash[0..8]);

    // Some(authority)
    data[8] = 1; // Some variant
    data[9..41].copy_from_slice(authority.as_ref());

    let rent = svm.minimum_balance_for_rent_exemption(data.len());

    svm.set_account(
        pk_to_addr(&pda),
        Account {
            lamports: rent,
            data,
            owner: pk_to_addr(&program_id()).0.into(),
            executable: false,
            rent_epoch: 0,
        },
    ).unwrap();

    (pk_to_addr(&pda), bump)
}

/// Create whitelist entry PDA
fn create_whitelist_entry(
    svm: &mut LiteSVM,
    address: &Pubkey,
) -> (Address, u8) {
    let (pda, bump) = Pubkey::find_program_address(
        &[WHITELIST_SEED, address.as_ref()],
        &program_id()
    );

    // WhitelistEntry: discriminator (8) + Pubkey (32) + i64 (8) = 48 bytes
    let mut data = vec![0u8; 48];

    // Anchor discriminator for WhitelistEntry
    let mut hasher = Sha256::new();
    hasher.update(b"account:WhitelistEntry");
    let hash = hasher.finalize();
    data[0..8].copy_from_slice(&hash[0..8]);

    // address field
    data[8..40].copy_from_slice(address.as_ref());

    // created_at (i64 timestamp)
    let timestamp: i64 = 1234567890;
    data[40..48].copy_from_slice(&timestamp.to_le_bytes());

    let rent = svm.minimum_balance_for_rent_exemption(data.len());

    svm.set_account(
        pk_to_addr(&pda),
        Account {
            lamports: rent,
            data,
            owner: pk_to_addr(&program_id()).0.into(),
            executable: false,
            rent_epoch: 0,
        },
    ).unwrap();

    (pk_to_addr(&pda), bump)
}

/// Create ExtraAccountMetaList PDA for a mint
fn create_extra_account_meta_list(
    svm: &mut LiteSVM,
    mint: &Pubkey,
) -> (Address, u8) {
    use spl_tlv_account_resolution::{account::ExtraAccountMeta, state::ExtraAccountMetaList};
    use spl_tlv_account_resolution::seeds::Seed;

    let (pda, bump) = Pubkey::find_program_address(
        &[EXTRA_ACCOUNT_METAS_SEED, mint.as_ref()],
        &program_id()
    );

    // Build the same metas as initialize_extra_account_meta_list
    let extra_metas = vec![
        // Whitelist PDA for source (index 0 = source_token_account)
        ExtraAccountMeta::new_with_seeds(
            &[
                Seed::Literal { bytes: WHITELIST_SEED.to_vec() },
                Seed::AccountKey { index: 0 }, // source_token_account
            ],
            false, // is_signer
            false, // is_writable
        ).unwrap(),
        // Whitelist PDA for destination (index 2 = destination_token_account)
        ExtraAccountMeta::new_with_seeds(
            &[
                Seed::Literal { bytes: WHITELIST_SEED.to_vec() },
                Seed::AccountKey { index: 2 }, // destination_token_account
            ],
            false,
            false,
        ).unwrap(),
    ];

    let space = ExtraAccountMetaList::size_of(extra_metas.len()).unwrap();
    let mut data = vec![0u8; space];

    ExtraAccountMetaList::init::<ExecuteInstructionDummy>(&mut data, &extra_metas).unwrap();

    let rent = svm.minimum_balance_for_rent_exemption(space);

    svm.set_account(
        pk_to_addr(&pda),
        Account {
            lamports: rent,
            data,
            owner: pk_to_addr(&program_id()).0.into(),
            executable: false,
            rent_epoch: 0,
        },
    ).unwrap();

    (pk_to_addr(&pda), bump)
}
