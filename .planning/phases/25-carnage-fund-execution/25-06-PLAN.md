```rust
pub fn handler<'info>(
    ctx: Context<'_, '_, 'info, 'info, ExecuteCarnage<'info>>,
) -> Result<()> {
    let clock = Clock::get()?;

    // Validate deadline hasn't expired
    require!(
        clock.slot <= ctx.accounts.epoch_state.carnage_deadline_slot,
        EpochError::CarnageDeadlineExpired
    );

    // Read pending action and target from EXISTING EpochState fields
    let action = CarnageAction::from_u8(ctx.accounts.epoch_state.carnage_action)
        .ok_or(EpochError::InvalidCarnageTargetPool)?;
    let target = Token::from_u8(ctx.accounts.epoch_state.carnage_target)
        .ok_or(EpochError::InvalidCarnageTargetPool)?;
    let current_epoch = ctx.accounts.epoch_state.current_epoch;

    // Validate target pool matches pending target
    let expected_mint = match target {
        Token::Crime => ctx.accounts.crime_vault.mint,
        Token::Fraud => ctx.accounts.fraud_vault.mint,
    };
    require!(
        ctx.accounts.mint_b.key() == expected_mint,
        EpochError::InvalidCarnageTargetPool
    );

    let mut tokens_burned: u64 = 0;
    let mut sol_from_sale: u64 = 0;

    // Build signer seeds for carnage_signer PDA
    let carnage_signer_bump = ctx.bumps.carnage_signer;
    let carnage_signer_seeds: &[&[u8]] = &[CARNAGE_SIGNER_SEED, &[carnage_signer_bump]];

    // Build signer seeds for carnage_state PDA (authority for token vaults)
    let carnage_state_bump = ctx.accounts.carnage_state.bump;
    let carnage_state_seeds: &[&[u8]] = &[CARNAGE_FUND_SEED, &[carnage_state_bump]];

    // Read held token info before mutable operations
    let held_amount = ctx.accounts.carnage_state.held_amount;
    let held_token = ctx.accounts.carnage_state.held_token;

    // === Step 1: Handle existing holdings ===
    match action {
        CarnageAction::Burn => {
            if held_amount > 0 {
                let carnage_state = &mut ctx.accounts.carnage_state;
                tokens_burned = burn_held_tokens(
                    carnage_state,
                    &ctx.accounts.crime_vault,
                    &ctx.accounts.fraud_vault,
                    &ctx.accounts.token_program_b,
                    carnage_state_seeds,
                )?;
            }
        }
        CarnageAction::Sell => {
            if held_amount > 0 {
                sol_from_sale = execute_sell_swap(
                    &ctx,
                    held_amount,
                    held_token,
                    carnage_signer_seeds,
                )?;

                let carnage_state = &mut ctx.accounts.carnage_state;
                carnage_state.held_token = 0;
                carnage_state.held_amount = 0;
            }
        }
        CarnageAction::None => {
            // BuyOnly path - no holdings to handle
        }
    }

    // === Step 2: Buy target token ===
    let sol_balance = ctx.accounts.sol_vault.lamports();
    let swap_amount = std::cmp::min(sol_balance, MAX_CARNAGE_SWAP_LAMPORTS);

    let tokens_bought = if swap_amount > 0 {
        execute_buy_swap(&ctx, swap_amount, target, carnage_signer_seeds)?
    } else {
        0
    };

    // === Step 3: Update state ===
    let carnage_state = &mut ctx.accounts.carnage_state;
    let epoch_state = &mut ctx.accounts.epoch_state;

    carnage_state.held_token = target.to_u8() + 1;
    carnage_state.held_amount = tokens_bought;
    carnage_state.total_sol_spent = carnage_state
        .total_sol_spent
        .checked_add(swap_amount)
        .ok_or(EpochError::Overflow)?;
    carnage_state.total_triggers = carnage_state
        .total_triggers
        .checked_add(1)
        .ok_or(EpochError::Overflow)?;
    carnage_state.last_trigger_epoch = current_epoch;

    epoch_state.carnage_pending = false;
    epoch_state.carnage_action = CarnageAction::None.to_u8();
    epoch_state.last_carnage_epoch = current_epoch;

    msg!(
        "Carnage executed (fallback): action={}, target={}, sol_spent={}, bought={}, burned={}",
        action.to_u8(),
        target.to_u8(),
        swap_amount,
        tokens_bought,
        tokens_burned
    );

    emit!(CarnageExecuted {
        epoch: current_epoch,
        action: action.to_u8(),
        target: target.to_u8(),
        sol_spent: swap_amount,
        tokens_bought,
        tokens_burned,
        sol_from_sale,
        atomic: false,  // This is fallback execution
    });

    Ok(())
}
```
