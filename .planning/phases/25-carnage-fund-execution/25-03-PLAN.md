//! execute_carnage_atomic instruction.
//!
//! Public instruction for Carnage execution bundled with consume_randomness.
//! Executes burn/sell + buy operations based on EpochState pending fields.
//!
//! CRITICAL CPI DEPTH: This path is exactly at Solana's limit:
//!   execute_carnage_atomic (entry) -> Tax::swap_exempt (1)
//!   -> AMM::swap_sol_pool (2) -> Token-2022::transfer_checked (3)
//!   -> Transfer Hook::execute (4) -- SOLANA LIMIT
//!
//! DO NOT add any CPI calls to the swap path.
//!
//! Source: Carnage_Fund_Spec.md Sections 8-10, 13.2

use anchor_lang::prelude::*;
use anchor_lang::solana_program::{
    instruction::{AccountMeta, Instruction},
    program::invoke_signed,
};
use anchor_spl::token_interface::{Mint, TokenAccount, TokenInterface};

use crate::constants::{
    CARNAGE_FUND_SEED, CARNAGE_SIGNER_SEED, CARNAGE_SOL_VAULT_SEED,
    MAX_CARNAGE_SWAP_LAMPORTS, EPOCH_STATE_SEED,
};
use crate::errors::EpochError;
use crate::events::CarnageExecuted;
use crate::state::{CarnageFundState, CarnageAction, EpochState, Token};

/// Accounts for execute_carnage_atomic instruction.
///
/// This instruction is PUBLIC - can be called by anyone when carnage_pending = true.
/// Typically bundled with consume_randomness in the same transaction for MEV protection.
#[derive(Accounts)]
pub struct ExecuteCarnageAtomic<'info> {
    /// Caller (anyone - permissionless execution)
    pub caller: Signer<'info>,

    /// Global epoch state (has pending Carnage flags)
    /// NOTE: EpochState already has carnage_pending, carnage_action, carnage_target,
    /// carnage_deadline_slot fields from Phase 23 - we READ these existing fields.
    #[account(
        mut,
        seeds = [EPOCH_STATE_SEED],
        bump = epoch_state.bump,
        constraint = epoch_state.initialized @ EpochError::NotInitialized,
        constraint = epoch_state.carnage_pending @ EpochError::NoCarnagePending,
    )]
    pub epoch_state: Account<'info, EpochState>,

    /// Carnage Fund state
    #[account(
        mut,
        seeds = [CARNAGE_FUND_SEED],
        bump = carnage_state.bump,
        constraint = carnage_state.initialized @ EpochError::CarnageNotInitialized,
    )]
    pub carnage_state: Account<'info, CarnageFundState>,

    /// Carnage signer PDA - signs Tax::swap_exempt CPI
    /// CHECK: PDA derived from known seeds, used as signer for swap_exempt
    #[account(
        seeds = [CARNAGE_SIGNER_SEED],
        bump,
    )]
    pub carnage_signer: AccountInfo<'info>,

    /// Carnage SOL vault (holds native SOL as lamports)
    /// CHECK: PDA derived from known seeds, holds native SOL
    #[account(
        mut,
        seeds = [CARNAGE_SOL_VAULT_SEED],
        bump,
    )]
    pub sol_vault: AccountInfo<'info>,

    /// Carnage's WSOL token account (for swap_exempt user_token_a)
    /// This receives wrapped SOL before swap and unwraps after
    #[account(mut)]
    pub carnage_wsol: InterfaceAccount<'info, TokenAccount>,

    /// Carnage CRIME vault (Token-2022 account)
    #[account(
        mut,
        constraint = crime_vault.key() == carnage_state.crime_vault @ EpochError::InvalidCarnageTargetPool,
    )]
    pub crime_vault: InterfaceAccount<'info, TokenAccount>,

    /// Carnage FRAUD vault (Token-2022 account)
    #[account(
        mut,
        constraint = fraud_vault.key() == carnage_state.fraud_vault @ EpochError::InvalidCarnageTargetPool,
    )]
    pub fraud_vault: InterfaceAccount<'info, TokenAccount>,

    // === Target Pool Accounts (CRIME/SOL or FRAUD/SOL) ===

    /// AMM pool for target token (CRIME/SOL or FRAUD/SOL)
    /// CHECK: Validated in Tax::swap_exempt CPI
    #[account(mut)]
    pub target_pool: AccountInfo<'info>,

    /// Pool's SOL vault
    #[account(mut)]
    pub pool_vault_a: InterfaceAccount<'info, TokenAccount>,

    /// Pool's IP token vault (CRIME or FRAUD)
    #[account(mut)]
    pub pool_vault_b: InterfaceAccount<'info, TokenAccount>,

    /// WSOL mint
    pub mint_a: InterfaceAccount<'info, Mint>,

    /// Target token mint (CRIME or FRAUD, Token-2022)
    pub mint_b: InterfaceAccount<'info, Mint>,

    // === Programs ===

    /// Tax Program (for swap_exempt CPI)
    /// CHECK: Program ID validated during CPI
    pub tax_program: AccountInfo<'info>,

    /// AMM Program (passed to Tax for swap)
    /// CHECK: Program ID validated during CPI
    pub amm_program: AccountInfo<'info>,

    /// SPL Token program (for WSOL)
    pub token_program_a: Interface<'info, TokenInterface>,

    /// Token-2022 program (for CRIME/FRAUD)
    pub token_program_b: Interface<'info, TokenInterface>,

    /// System program
    pub system_program: Program<'info, System>,
}

/// Handler for execute_carnage_atomic instruction.
///
/// # Flow
/// 1. Read pending action and target from EpochState (existing fields)
/// 2. If action is Burn: burn all held tokens
/// 3. If action is Sell: sell held tokens to SOL via swap_exempt
/// 4. Calculate swap amount (min of balance and 1000 SOL cap)
/// 5. Buy target token via swap_exempt
/// 6. Update CarnageFundState holdings and statistics
/// 7. Clear carnage_pending flag
/// 8. Emit CarnageExecuted event
pub fn handler<'info>(
    ctx: Context<'_, '_, 'info, 'info, ExecuteCarnageAtomic<'info>>,
) -> Result<()> {
    let epoch_state = &mut ctx.accounts.epoch_state;
    let carnage_state = &mut ctx.accounts.carnage_state;
    let clock = Clock::get()?;

    // Read pending action and target from EXISTING EpochState fields
    let action = CarnageAction::from_u8(epoch_state.carnage_action)
        .ok_or(EpochError::InvalidCarnageTargetPool)?;
    let target = Token::from_u8(epoch_state.carnage_target)
        .ok_or(EpochError::InvalidCarnageTargetPool)?;

    // Validate target pool matches pending target
    let expected_mint = match target {
        Token::Crime => ctx.accounts.crime_vault.mint,
        Token::Fraud => ctx.accounts.fraud_vault.mint,
    };
    require!(
        ctx.accounts.mint_b.key() == expected_mint,
        EpochError::InvalidCarnageTargetPool
    );

    let mut tokens_burned: u64 = 0;
    let mut sol_from_sale: u64 = 0;

    // Build signer seeds for carnage_signer PDA
    let carnage_signer_bump = ctx.bumps.carnage_signer;
    let carnage_signer_seeds: &[&[u8]] = &[
        CARNAGE_SIGNER_SEED,
        &[carnage_signer_bump],
    ];

    // === Step 1: Handle existing holdings ===
    match action {
        CarnageAction::Burn => {
            if carnage_state.held_amount > 0 {
                tokens_burned = burn_held_tokens(
                    carnage_state,
                    &ctx.accounts.crime_vault,
                    &ctx.accounts.fraud_vault,
                    &ctx.accounts.token_program_b,
                    &ctx.accounts.carnage_state.to_account_info(),
                    &[CARNAGE_FUND_SEED, &[carnage_state.bump]],
                )?;
            }
        }
        CarnageAction::Sell => {
            if carnage_state.held_amount > 0 {
                // Sell held tokens -> SOL via swap_exempt
                sol_from_sale = execute_sell_swap(
                    &ctx,
                    carnage_state,
                    carnage_signer_seeds,
                )?;
            }
        }
        CarnageAction::None => {
            // BuyOnly path - no holdings to handle
        }
    }

    // === Step 2: Buy target token ===
    // Calculate swap amount: min of SOL balance and cap
    let sol_balance = ctx.accounts.sol_vault.lamports();
    let swap_amount = std::cmp::min(sol_balance, MAX_CARNAGE_SWAP_LAMPORTS);

    let tokens_bought = if swap_amount > 0 {
        execute_buy_swap(
            &ctx,
            swap_amount,
            target,
            carnage_signer_seeds,
        )?
    } else {
        0
    };

    // === Step 3: Update state ===
    carnage_state.held_token = target.to_u8() + 1; // 1=CRIME, 2=FRAUD (0=None)
    carnage_state.held_amount = tokens_bought;
    carnage_state.total_sol_spent = carnage_state.total_sol_spent
        .checked_add(swap_amount)
        .ok_or(EpochError::Overflow)?;
    carnage_state.total_triggers = carnage_state.total_triggers
        .checked_add(1)
        .ok_or(EpochError::Overflow)?;
    carnage_state.last_trigger_epoch = epoch_state.current_epoch;

    // Clear pending flags (writing to EXISTING EpochState fields)
    epoch_state.carnage_pending = false;
    epoch_state.carnage_action = CarnageAction::None.to_u8();
    epoch_state.last_carnage_epoch = epoch_state.current_epoch;

    msg!(
        "Carnage executed: action={}, target={}, sol_spent={}, bought={}, burned={}",
        action.to_u8(),
        target.to_u8(),
        swap_amount,
        tokens_bought,
        tokens_burned
    );

    emit!(CarnageExecuted {
        epoch: epoch_state.current_epoch,
        action: action.to_u8(),
        target: target.to_u8(),
        sol_spent: swap_amount,
        tokens_bought,
        tokens_burned,
        sol_from_sale,
        atomic: true,
    });

    Ok(())
}

/// Burn held tokens from the appropriate vault.
///
/// Token-2022 burns do NOT trigger transfer hooks, so this doesn't
/// consume CPI depth for the hook call.
fn burn_held_tokens<'info>(
    carnage_state: &mut Account<'info, CarnageFundState>,
    crime_vault: &InterfaceAccount<'info, TokenAccount>,
    fraud_vault: &InterfaceAccount<'info, TokenAccount>,
    token_program: &Interface<'info, TokenInterface>,
    authority: &AccountInfo<'info>,
    signer_seeds: &[&[u8]],
) -> Result<u64> {
    let amount = carnage_state.held_amount;
    if amount == 0 {
        return Ok(0);
    }

    // Determine which vault to burn from
    let (vault, is_crime) = match carnage_state.held_token {
        1 => (crime_vault.to_account_info(), true),  // CRIME
        2 => (fraud_vault.to_account_info(), false), // FRAUD
        _ => return Ok(0), // No holdings
    };

    // Build burn instruction
    let burn_ix = spl_token_2022::instruction::burn(
        token_program.key,
        vault.key,
        &if is_crime { crime_vault.mint } else { fraud_vault.mint },
        authority.key,
        &[],
        amount,
    )?;

    invoke_signed(
        &burn_ix,
        &[vault, authority.clone(), token_program.to_account_info()],
        &[signer_seeds],
    )?;

    // Update statistics
    if is_crime {
        carnage_state.total_crime_burned = carnage_state.total_crime_burned
            .checked_add(amount)
            .ok_or(EpochError::Overflow)?;
    } else {
        carnage_state.total_fraud_burned = carnage_state.total_fraud_burned
            .checked_add(amount)
            .ok_or(EpochError::Overflow)?;
    }

    // Clear holdings
    carnage_state.held_token = 0;
    carnage_state.held_amount = 0;

    msg!("Burned {} tokens from {} vault", amount, if is_crime { "CRIME" } else { "FRAUD" });

    Ok(amount)
}

/// Execute sell swap: held token -> SOL via Tax::swap_exempt.
fn execute_sell_swap<'info>(
    ctx: &Context<'_, '_, 'info, 'info, ExecuteCarnageAtomic<'info>>,
    carnage_state: &mut Account<'info, CarnageFundState>,
    carnage_signer_seeds: &[&[u8]],
) -> Result<u64> {
    let amount = carnage_state.held_amount;
    if amount == 0 {
        return Ok(0);
    }

    // Determine source vault based on held token
    let (source_vault, is_crime) = match carnage_state.held_token {
        1 => (&ctx.accounts.crime_vault, true),
        2 => (&ctx.accounts.fraud_vault, false),
        _ => return Ok(0),
    };

    // Build swap_exempt CPI for sell (direction = 1 = BtoA)
    execute_swap_exempt_cpi(
        ctx,
        amount,
        1, // BtoA (sell token for SOL)
        is_crime,
        carnage_signer_seeds,
    )?;

    // Clear holdings (SOL received goes to sol_vault)
    carnage_state.held_token = 0;
    carnage_state.held_amount = 0;

    // Return amount as placeholder - actual SOL received is in vault
    // Note: In production, we'd read the vault balance difference
    Ok(amount)
}

/// Execute buy swap: SOL -> target token via Tax::swap_exempt.
fn execute_buy_swap<'info>(
    ctx: &Context<'_, '_, 'info, 'info, ExecuteCarnageAtomic<'info>>,
    amount: u64,
    target: Token,
    carnage_signer_seeds: &[&[u8]],
) -> Result<u64> {
    if amount == 0 {
        return Ok(0);
    }

    let is_crime = matches!(target, Token::Crime);

    // Build swap_exempt CPI for buy (direction = 0 = AtoB)
    execute_swap_exempt_cpi(
        ctx,
        amount,
        0, // AtoB (buy token with SOL)
        is_crime,
        carnage_signer_seeds,
    )?;

    // Return amount as placeholder for tokens received
    // Note: In production, we'd read the vault balance difference
    Ok(amount)
}

/// Execute Tax::swap_exempt CPI.
///
/// This function builds and executes the CPI to Tax Program's swap_exempt.
/// The swap_exempt instruction validates carnage_signer is from Epoch Program.
fn execute_swap_exempt_cpi<'info>(
    ctx: &Context<'_, '_, 'info, 'info, ExecuteCarnageAtomic<'info>>,
    amount: u64,
    direction: u8,
    is_crime: bool,
    carnage_signer_seeds: &[&[u8]],
) -> Result<()> {
    // Anchor discriminator for swap_exempt: sha256("global:swap_exempt")[0..8]
    const SWAP_EXEMPT_DISCRIMINATOR: [u8; 8] = [0xf3, 0x5b, 0x9e, 0x48, 0xd3, 0x8a, 0x1c, 0x27];

    // Determine user_token_b based on target
    let user_token_b = if is_crime {
        &ctx.accounts.crime_vault
    } else {
        &ctx.accounts.fraud_vault
    };

    // Build account metas matching SwapExempt struct
    let mut account_metas = vec![
        AccountMeta::new_readonly(ctx.accounts.carnage_signer.key(), true), // carnage_authority (signer)
        // swap_authority is derived in Tax Program, we pass our carnage_signer
        AccountMeta::new(ctx.accounts.target_pool.key(), false),            // pool
        AccountMeta::new(ctx.accounts.pool_vault_a.key(), false),           // pool_vault_a
        AccountMeta::new(ctx.accounts.pool_vault_b.key(), false),           // pool_vault_b
        AccountMeta::new_readonly(ctx.accounts.mint_a.key(), false),        // mint_a (WSOL)
        AccountMeta::new_readonly(ctx.accounts.mint_b.key(), false),        // mint_b (IP token)
        AccountMeta::new(ctx.accounts.carnage_wsol.key(), false),           // user_token_a
        AccountMeta::new(user_token_b.key(), false),                        // user_token_b
        AccountMeta::new_readonly(ctx.accounts.amm_program.key(), false),   // amm_program
        AccountMeta::new_readonly(ctx.accounts.token_program_a.key(), false), // token_program_a
        AccountMeta::new_readonly(ctx.accounts.token_program_b.key(), false), // token_program_b
        AccountMeta::new_readonly(ctx.accounts.system_program.key(), false),  // system_program
    ];

    // Add remaining_accounts for transfer hook support
    for account in ctx.remaining_accounts.iter() {
        if account.is_writable {
            account_metas.push(AccountMeta::new(account.key(), account.is_signer));
        } else {
            account_metas.push(AccountMeta::new_readonly(account.key(), account.is_signer));
        }
    }

    // Build instruction data: discriminator + amount_in + direction + is_crime
    let mut ix_data = Vec::with_capacity(18);
    ix_data.extend_from_slice(&SWAP_EXEMPT_DISCRIMINATOR);
    ix_data.extend_from_slice(&amount.to_le_bytes());
    ix_data.push(direction);
    ix_data.push(if is_crime { 1 } else { 0 });

    let ix = Instruction {
        program_id: ctx.accounts.tax_program.key(),
        accounts: account_metas,
        data: ix_data,
    };

    // Build account infos
    let mut account_infos = vec![
        ctx.accounts.carnage_signer.to_account_info(),
        ctx.accounts.target_pool.to_account_info(),
        ctx.accounts.pool_vault_a.to_account_info(),
        ctx.accounts.pool_vault_b.to_account_info(),
        ctx.accounts.mint_a.to_account_info(),
        ctx.accounts.mint_b.to_account_info(),
        ctx.accounts.carnage_wsol.to_account_info(),
        if is_crime {
            ctx.accounts.crime_vault.to_account_info()
        } else {
            ctx.accounts.fraud_vault.to_account_info()
        },
        ctx.accounts.amm_program.to_account_info(),
        ctx.accounts.token_program_a.to_account_info(),
        ctx.accounts.token_program_b.to_account_info(),
        ctx.accounts.system_program.to_account_info(),
    ];

    for account in ctx.remaining_accounts.iter() {
        account_infos.push(account.clone());
    }

    account_infos.push(ctx.accounts.tax_program.to_account_info());

    invoke_signed(&ix, &account_infos, &[carnage_signer_seeds])?;

    Ok(())
}
