  1. Update `TaxConfig` struct. The EpochState already stores individual per-token fields (`crime_buy_tax_bps`, etc.), so the legacy `low_tax_bps`/`high_tax_bps` fields in TaxConfig are kept for backward compatibility but set to 0 (superseded by per-token rates). **Do NOT add new fields or remove existing fields** -- just change what values are assigned:

     ```rust
     pub struct TaxConfig {
         pub cheap_side: Token,
         /// Legacy: Set to 0 (superseded by per-token rates below)
         pub low_tax_bps: u16,
         /// Legacy: Set to 0 (superseded by per-token rates below)
         pub high_tax_bps: u16,
         pub crime_buy_tax_bps: u16,
         pub crime_sell_tax_bps: u16,
         pub fraud_buy_tax_bps: u16,
         pub fraud_sell_tax_bps: u16,
     }
     ```

  2. Rewrite `derive_taxes` function to use 4 independent magnitude bytes:
     ```rust
     pub fn derive_taxes(vrf_result: &[u8; 32], current_cheap: Token) -> TaxConfig {
         // Byte 0: Flip decision (unchanged)
         let should_flip = vrf_result[0] < FLIP_THRESHOLD;

         // Byte 1: CRIME low tax magnitude (independent)
         let crime_low_bps = LOW_RATES[(vrf_result[1] % 4) as usize];

         // Byte 2: CRIME high tax magnitude (independent)
         let crime_high_bps = HIGH_RATES[(vrf_result[2] % 4) as usize];

         // Byte 3: FRAUD low tax magnitude (independent)
         let fraud_low_bps = LOW_RATES[(vrf_result[3] % 4) as usize];

         // Byte 4: FRAUD high tax magnitude (independent)
         let fraud_high_bps = HIGH_RATES[(vrf_result[4] % 4) as usize];

         let cheap_side = if should_flip {
             current_cheap.opposite()
         } else {
             current_cheap
         };

         // Assign rates: cheap side gets its OWN low buy / high sell
         // Expensive side gets its OWN high buy / low sell
         let (crime_buy, crime_sell, fraud_buy, fraud_sell) = match cheap_side {
             Token::Crime => (
                 crime_low_bps,   // CRIME cheap: low buy
                 crime_high_bps,  // CRIME cheap: high sell
                 fraud_high_bps,  // FRAUD expensive: high buy
                 fraud_low_bps,   // FRAUD expensive: low sell
             ),
             Token::Fraud => (
                 crime_high_bps,  // CRIME expensive: high buy
                 crime_low_bps,   // CRIME expensive: low sell
                 fraud_low_bps,   // FRAUD cheap: low buy
                 fraud_high_bps,  // FRAUD cheap: high sell
             ),
         };

         TaxConfig {
             cheap_side,
             low_tax_bps: 0,   // Legacy field, superseded by per-token rates
             high_tax_bps: 0,  // Legacy field, superseded by per-token rates
             crime_buy_tax_bps: crime_buy,
             crime_sell_tax_bps: crime_sell,
             fraud_buy_tax_bps: fraud_buy,
             fraud_sell_tax_bps: fraud_sell,
         }
     }
     ```
